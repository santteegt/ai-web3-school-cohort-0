import { ethers } from 'ethers';

// ERC-8004 registry contract ABI (from spec or deployed instance)
const ERC8004_ABI = [
  "function register(string name, bytes32 capability, uint256 credibility) returns (uint256 agentId)",
  "function getAgentInfo(uint256 agentId) returns (address owner, string name, bytes32 capability, uint256 credibility, uint256 createdAt)"
];

const BASE_MAINNET_RPC = process.env.BASE_MAINNET_RPC!;
const ERC8004_REGISTRY = process.env.ERC8004_REGISTRY_ADDRESS!; // deployed contract on Base
const ORCHESTRATOR_KEY = process.env.ORCHESTRATOR_PRIVATE_KEY!;
const SPECIALIST_KEY = process.env.SPECIALIST_PRIVATE_KEY!;

async function registerAgent(name: string, capability: string, privateKey: string) {
  const provider = new ethers.JsonRpcProvider(BASE_MAINNET_RPC);
  const signer = new ethers.Wallet(privateKey, provider);
  
  const registry = new ethers.Contract(ERC8004_REGISTRY, ERC8004_ABI, signer);
  
  console.log(`[Register] ${name} at address ${signer.address}`);
  
  const tx = await registry.register(
    name,
    ethers.encodeBytes32String(capability), // e.g., "task_execution"
    1000 // credibility score (adjust as needed)
  );
  
  const receipt = await tx.wait();
  console.log(`[Register] ${name} registered: tx=${receipt.hash}`);
  
  // Query back agent info
  const agentId = receipt.logs[0]; // parse event or use multicall
  const agentInfo = await registry.getAgentInfo(agentId);
  console.log(`[Register] Agent info:`, agentInfo);
  
  return { agentId, txHash: receipt.hash, agentInfo };
}

// Main
(async () => {
  const orchestrator = await registerAgent(
    "GuildOS-Orchestrator",
    "proposal_orchestration",
    ORCHESTRATOR_KEY
  );
  console.log("[Main] Orchestrator registered:", orchestrator);
  
  const specialist = await registerAgent(
    "GuildOS-Specialist",
    "task_execution",
    SPECIALIST_KEY
  );
  console.log("[Main] Specialist registered:", specialist);
})();
```

import axios from 'axios';

const A2A_BROKER = process.env.A2A_BROKER_URL!; // e.g., "https://a2a.example.com"
const ORCHESTRATOR_ADDR = "0x..."; // your registered orchestrator address
const SPECIALIST_ADDR = "0x..."; // your registered specialist address

interface TaskQuoteRequest {
  agent_id: string;
  task_type: string;
  metadata: Record<string, any>;
}

async function requestTaskQuote(request: TaskQuoteRequest) {
  console.log(`[A2A Gate 0] Sending quote request from ${ORCHESTRATOR_ADDR} to ${SPECIALIST_ADDR}`);
  
  const response = await axios.post(`${A2A_BROKER}/task/quote`, {
    sender: ORCHESTRATOR_ADDR,
    receiver: SPECIALIST_ADDR,
    payload: request,
    timestamp: Math.floor(Date.now() / 1000),
  });
  
  console.log(`[A2A Gate 0] Quote received:`, response.data);
  return response.data;
}

// Main: Request a quote for a sample task
(async () => {
  const quote = await requestTaskQuote({
    agent_id: SPECIALIST_ADDR,
    task_type: "write_documentation",
    metadata: {
      expected_output: "README.md for GuildOS",
      reward_eth: "0.01",
      deadline_hours: 24,
    },
  });
  
  console.log("[Main] Quote accepted. Next step: send task via Gate 1 (task/send).");
})();
```
